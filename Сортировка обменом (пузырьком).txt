def bubble_sort(arr):
    # Получаем длину массива
    n = len(arr)
    
    # Внешний цикл выполняется столько раз, сколько элементов в массиве минус один
    for i in range(n - 1):
        # Флаг для отслеживания произведенных перестановок
        swapped = False
        
        # Внутренний цикл идет от первого элемента до конца массива, уменьшенного на i
        # (так как на предыдущем проходе наибольший элемент гарантированно окажется на своей позиции)
        for j in range(0, n-i-1):
            # Если текущий элемент больше следующего, производим обмен мест
            if arr[j] > arr[j+1]:
                # Меняем элементы местами с помощью простого присваивания
                arr[j], arr[j+1] = arr[j+1], arr[j]
                
                # Устанавливаем флаг True, чтобы обозначить наличие перестановки
                swapped = True
        
        # Если на текущем проходе не было сделано ни одной перестановки, массив уже отсортирован
        if not swapped:
            break

# Пример использования
if __name__ == "__main__":
    # Массив для сортировки
    array = [64, 34, 25, 12, 22, 11, 90]
    
    # Выводим исходный массив
    print("Исходный массив:", array)
    
    # Применяем сортировку пузырьком
    bubble_sort(array)
    
    # Выводим отсортированный массив
    print("Отсортированный массив:", array)

Пример выполнения:
Исходный массив: [64, 34, 25, 12, 22, 11, 90]
Отсортированный массив: [11, 12, 22, 25, 34, 64, 90]
