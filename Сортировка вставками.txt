def insertion_sort(arr):
    """
    Реализация алгоритма сортировки вставками.
    """
    # Получаем длину массива
    n = len(arr)
    
    # Начинаем проход с второго элемента (индекс 1), так как первый элемент считается уже отсортированным
    for i in range(1, n):
        key = arr[i]          # Берём текущий элемент, который будем вставлять на нужную позицию
        j = i - 1             # Индекс предыдущего элемента
        
        # Пока предыдущий элемент больше нашего текущего ключа и мы ещё не дошли до начала массива
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]  # Сдвигаем больший элемент вправо
            j -= 1                # Переходим к следующему слева элементу
            
        # Наконец, вставляем наш ключ на освобожденное место
        arr[j + 1] = key

# Пример использования
if __name__ == "__main__":
    # Исходный массив
    array = [12, 11, 13, 5, 6]
    
    # Выводим исходный массив
    print("Исходный массив:", array)
    
    # Сортируем массив
    insertion_sort(array)
    
    # Выводим отсортированный массив
    print("Отсортированный массив:", array)

Пример выполнения:
Исходный массив: [12, 11, 13, 5, 6]
Отсортированный массив: [5, 6, 11, 12, 13]
